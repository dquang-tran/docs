FleetWorks sends bid events to your TMS when carriers submit bids on loads. The webhook payload contains bid details and carrier information.

<Note>
  Configure your webhook URL in the FleetWorks dashboard to receive bid events.
</Note>

## Bid Event Payload

<ParamField path="loadId" type="string" required>
  Load identifier
</ParamField>

<ParamField path="amount" type="number" required>
  Bid amount
</ParamField>

<ParamField path="notes" type="string">
  Additional bid details
</ParamField>

<ParamField path="url" type="string">
  Link to bid on FleetWorks platform
</ParamField>

<ParamField path="source" type="object" required>
  Source of the bid
  <Expandable>
    <ParamField path="type" type="string" required>
      Type of source ("call" or "email")
    </ParamField>
    <ParamField path="id" type="string" required>
      Unique identifier of the source (call ID or email thread ID)
    </ParamField>
    <ParamField path="direction" type="string" required>
      Direction of communication ("inbound" or "outbound")
    </ParamField>
    <ParamField path="initiator" type="object" required>
      Party that initiated the communication
      <Expandable>
        <ParamField path="type" type="string" required>
          Type of party ("carrier" or "broker")
        </ParamField>
        <ParamField path="phone" type="object">
          Phone contact details
          <Expandable>
            <ParamField path="number" type="string">
              Phone number in E.164 format
            </ParamField>
            <ParamField path="extension" type="string">
              Phone extension
            </ParamField>
          </Expandable>
        </ParamField>
        <ParamField path="email" type="string">
          Email address
        </ParamField>
        <ParamField path="name" type="string">
          Name of the initiating party
        </ParamField>
      </Expandable>
    </ParamField>
    <ParamField path="receiver" type="object" required>
      Party that received the communication
      <Expandable>
        <ParamField path="type" type="string" required>
          Type of party ("carrier" or "broker")
        </ParamField>
        <ParamField path="phone" type="object">
          Phone contact details
          <Expandable>
            <ParamField path="number" type="string">
              Phone number in E.164 format

            <ParamField path="extension" type="string">
              Phone extension
            </ParamField>
          </Expandable>
        </ParamField>
        <ParamField path="email" type="string">
          Email address
        </ParamField>
        <ParamField path="name" type="string">
          Name of the receiving party
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

      Carrier email address
    </ParamField>
  </Expandable>
